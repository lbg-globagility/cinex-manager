
CREATE TABLE IF NOT EXISTS `session_ewallet` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `session_id` varchar(100) DEFAULT NULL,
  `contact_number` varchar(100) DEFAULT NULL,
  `reference_number` varchar(50) DEFAULT NULL,
  `ewallet_method` enum('GCASH','PAYMAYA') DEFAULT 'PAYMAYA',
  PRIMARY KEY (`id`),
  KEY `session_id` (`session_id`),
  CONSTRAINT `FK__session` FOREIGN KEY (`session_id`) REFERENCES `session` (`session_id`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1;

	ADD CONSTRAINT `FK_session_ewallet_ewallets` FOREIGN KEY (`ewallet_id`) REFERENCES `ewallets` (`id`) ON UPDATE RESTRICT ON DELETE RESTRICT;
